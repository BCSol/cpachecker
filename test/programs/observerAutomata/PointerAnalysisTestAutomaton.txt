AUTOMATON PointerAnalysisTestAutomaton
/* Queries the PointerAnalysis for errors and prints them. 
 * Does not abort the checking (does not go to the error state).
 */

INITIAL STATE Init;

STATE Init : 
  CHECK(PointerAnalysis("DOUBLE_FREE")) -> PRINT "Found a double Free" GOTO Init;
  CHECK(PointerAnalysis("MemoryLeak")) -> PRINT "Found a Memory Leak" GOTO Init;
  CHECK(PointerAnalysis("UNSAFE_DEREFERENCE")) -> PRINT "Found an UNSAFE_DEREFERENCE" GOTO Init;
  CHECK(PointerAnalysis("POTENTIALLY_UNSAFE_DEREFERENCE")) -> PRINT "Found a POTENTIALLY_UNSAFE_DEREFERENCE" GOTO Init;
  CHECK(PointerAnalysis("INVALID_FREE")) -> PRINT "Found an INVALID_FREE" GOTO Init;
