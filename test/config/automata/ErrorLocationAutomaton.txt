AUTOMATON ErrorLocationAutomaton
// This Automaton replaces the ErrorLocationCPA. 
//   It detects error locations that can be specified by arbitrary labels 
//   (or other pieces of matchable code) that occur in the source code.

/* The ErrorLocationCPA had 3 triggers.
 * The Trigger "(CFAEdge.CFASuccessor instanceOf CFAErrorNode)" could not be implemented in this automaton.
 */

INITIAL STATE Init;

STATE Init :
       // this transition matches if the label of the successor CFA location starts with "error"
     MATCH LABEL [error.*]
       // matches if assert_fail is called without arguments
  || MATCH {__assert_fail()}
       // matches if assert_fail is called with at least one argument
  || MATCH {__assert_fail($?)}

  -> ERROR;
