OBSERVER AUTOMATON LockingAutomaton
// Kommentar
/* Kommentar */

INITIAL STATE Init;

STATE Init :
  MATCH  { x = $?; } -> ERROR;
  MATCH  { init(); }  -> GOTO Unlocked;
  MATCH  { lock($?); }  -> ERROR;
  MATCH  { unlock($?); }  -> ERROR;


STATE Locked :
  MATCH  { x = $?; } -> GOTO Locked; // unnecessary line
  MATCH  { init(); }  ->  ERROR;
  MATCH  { lock($?); }  ->  ERROR;
  MATCH  { unlock($?); }  -> GOTO Unlocked;
  TRUE -> GOTO Locked;


STATE Unlocked :
  MATCH  { x = $?; } -> ERROR;
  MATCH  { init(); }  ->  ERROR;
  MATCH  { lock($?); }  -> GOTO Locked;
  MATCH  { unlock($?); }  -> ERROR;
  TRUE -> GOTO Unlocked;

END AUTOMATON
